// elaborado por um estudante do curso básico de programação c#.

class Aluno
{
    //Atributos - criar
    private string nome;
    private double nota1;
    private double nota2;

    //Encapsulamento - nome
    public string Nome { get { return nome; } set { if (string.IsNullOrWhiteSpace(value)) { nome = "Desconhecido"; } else { nome = value; } } }

    //Encapsulamento - nota1
    public double Nota1 
    { get { return nota1; } set { if (value < 0 || value > 20) { nota1 = 0;return; } else { nota1 = value; } } }

    //Encapsulamento - nota2
    public double Nota2
    { get { return nota2; } set { if (value < 0 || value > 20) { nota2 = 0;return; } else { nota2 = value; } } }

    //Construtor

    public Aluno(string nome, double nota1, double nota2) { Nome = nome; Nota1 = nota1; Nota2 = nota2; }

    //Métodos - Criar
    public double Media() { return (nota1 + nota2) / 2; }

    public bool Aprovado() { if (Media() >= 10) { return true; } else { return false; } }

}
class Programa
{
    static void Main()
    {
        //Criação da lista do tipo da classe

        List<Aluno> alunos = new List<Aluno>();

        //Adicionar dados objectos à lista

        alunos.Add(new Aluno("Edson Manuel", 15, 16));
        alunos.Add(new Aluno("Jorge André", 17, 8));
        alunos.Add(new Aluno("Marta", 8, 9));

        //Percorrer a lista

        for (int i = 0;i < alunos.Count; i++) { Aluno a = alunos[i]; Console.WriteLine($"Nome: {a.Nome}"); Console.WriteLine($"Nota 1: {a.Nota1}"); Console.WriteLine($"Nota 2: {a.Nota2}"); Console.WriteLine($"Média: {a.Media()}"); Console.WriteLine($"Aprovado? {a.Aprovado()}"); Console.WriteLine("---------------"); }
    }
}
